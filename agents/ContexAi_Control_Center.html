cd C:\Users\amirw\contexai-website

$bytes = New-Object byte[] 32
[Security.Cryptography.RNGCryptoServiceProvider]::Create().GetBytes($bytes)
($bytes | ForEach-Object { $_.ToString("x2") }) -join ""

npx wrangler secret put ADMIN_TOKEN
npx wrangler deploy
